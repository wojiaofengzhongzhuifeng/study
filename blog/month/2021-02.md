### 完成功能函数-某个数据初始化为 null，经过 x 秒后才设置值，如何保证使用该数据的时候已经设置好值了？

局限性太大

```javascript
  function test(data, key, fn, delay) {
    let timeId = null;
    if (!data[key]) {
      timeId = setInterval(() => {
        console.log('xxxxx')

        if (data[key]) {
          clearInterval(timeId);
          fn(data)
        }
      }, delay)
    }

  }


  let data = {};
  setTimeout(()=>{
    data.test1 = 1111
  }, 5000)
  test(data, 'test1',(newData)=>{
    console.log(newData)
  }, 1000)
```
