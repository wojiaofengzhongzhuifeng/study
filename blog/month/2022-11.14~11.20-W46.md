###  js 基础语法：如果想退出while 循环，使用break；如果想退出函数，使用return

```
function test1(){
    let index = 0
    while(index <= 3){
				index++
        if(index === 1){
            return index
        }
    }
    console.log("123321") // 不会打印这个
    return index
}
test1()
```

```
function test1(){
    let index = 0
    while(index <= 3){
				index++
        if(index === 1){
            break
        }
    }
    console.log("123321") // 会打印这个
    return index
}
test1()
```
