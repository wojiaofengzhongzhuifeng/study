  即从对象中根据 key 取 value 要特别小心，需要充分考虑以下两种情况

  - 没有这个 key
  - value不是从这个 key 得到

  ```
  // 数据可能情况1
  let obj = {
  	a: 1,
  }
  
  
  // 使用数据地方
  let number = obj.a
  
  // 这样很容易报错，因为假设有些数据返回是这样的话
  // 数据可能情况2
  let obj = {
  	A: 2,
  }
  
  // 这个时候应该这样处理
  let number = findValue(obj, 'a');
  
  function findValue(obj, key){
  	let value = obj[key] || obj.A// 在这里添加临时新增的
  }
  let number = findValue(obj, 'a')
  ```
